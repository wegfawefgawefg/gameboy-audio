How to code GameBoy music.

A Sequence Of Notes

To make a sequence of sounds you take the samples from one note, and the samples from another note, and you put them in the same list. 

output = []

for sample in first_note:
    output.append(sample)

for sample in second_note:
    output.append(sample)

You can also just make a list of notes, and loop through that list to render them. 

sample_rate = 44100

def square_sample(sample_index, frequency, volume):
    time = sample_index / sample_rate
    phase = (frequency * time) % 1
    if phase < 0.5:
        return volume
    return -volume

# heres ur list of notes
notes = [
    # frequency, duration
    (261.63, 0.20),  # C4
    (329.63, 0.20),  # E4
    (392.00, 0.20),  # G4
    (523.25, 0.30),  # C5
]

samples = []

# go through each note
for frequency, duration in notes:
    length = int(duration * sample_rate)

    for i in range(length):
        sample = square_sample(i, frequency, 0.25)
        samples.append(sample)

Notes can overlap.

You can play notes at the same time by adding the samples to each other. The total volume of two waves of vol 0.5 will end up being 1.0 bc its added on top, but its fine this is an okay way to mix notes for us.

You could clamp the number with min and max, but that changes the wave shape. Thats not clean mixing. Its more like distortion. For now the clean trick is boring: keep each note quiet enough that the added result does not go past -1.0 or 1.0.

Lets change our code so we can render a sequence but notes can overlap. So you do render_note and pass in the big list, and it will mix it into the existing list. 

sample_rate = 44100

def square(sample_index, frequency, volume):
    time = sample_index / sample_rate
    phase = (frequency * time) % 1
    if phase < 0.5:
        return volume
    return -volume


def render_note(output, start_time, duration, frequency, volume):
    start_index = int(start_time * sample_rate)
    num_samples = int(duration * sample_rate)

    for sample_index in range(num_samples):
        output_index = start_index + sample_index

        if output_index >= len(output):
            break

        note_sample = square(sample_index, frequency, volume)

        # += means mix this note into whatever is already there.
        output[output_index] += note_sample


notes = [
    # frequency, start time, duration, volume
    (220.00, 0.00, 0.65, 0.35),
    (330.00, 0.35, 0.65, 0.35),
]

# First, figure out how long the output list needs to be.
# Each note ends at its start time plus its duration.
# The note that ends farthest to the right decides the list size.
end_time = 0
for frequency, start_time, duration, volume in notes:
    note_end_time = start_time + duration
    if note_end_time > end_time:
        end_time = note_end_time

# Now turn that end time into the number of sample slots we need.
num_samples = int(end_time * sample_rate)
output = [0] * num_samples

for frequency, start_time, duration, volume in notes:
    render_note(output, start_time, duration, frequency, volume)

We can put the sample function in each note so that each note can use a different voice.

import math

sample_rate = 44100

def square(sample_index, frequency, volume):
    time = sample_index / sample_rate
    phase = (frequency * time) % 1
    if phase < 0.5:
        return volume
    return -volume


def sin(sample_index, frequency, volume):
    time = sample_index / sample_rate
    return math.sin(2 * math.pi * frequency * time) * volume


def triangle(sample_index, frequency, volume):
    time = sample_index / sample_rate
    phase = (frequency * time) % 1
    return (1 - 4 * abs(phase - 0.5)) * volume


def pulse(sample_index, frequency, volume):
    time = sample_index / sample_rate
    phase = (frequency * time) % 1
    if phase < 0.25:
        return volume
    return -volume


def bright(sample_index, frequency, volume):
    # A lead voice can be a mix of tiny waves.
    main = sin(sample_index, frequency, volume)
    sparkle = sin(sample_index, frequency * 2, volume * 0.35)
    edge = square(sample_index, frequency * 3, volume * 0.18)
    return main + sparkle + edge


def render_note(output, voice, frequency, start_time, duration, volume):
    start_index = int(start_time * sample_rate)
    num_samples = int(duration * sample_rate)

    for sample_index in range(num_samples):
        output_index = start_index + sample_index

        if output_index >= len(output):
            break

        output[output_index] += voice(sample_index, frequency, volume)

That already sounds pretty good and theres not that much code here. You also dont need a ton of voices to make something nice.

To make this a little nicer to use, make a lookup table of note letter number to frequency.

sample_rate = 44100

note_frequency = {
    "C4": 261.63,
    "D4": 293.66,
    "E4": 329.63,
    "G4": 392.00,
    "A4": 440.00,
    "C5": 523.25,
}

def square(sample_index, frequency, volume):
    time = sample_index / sample_rate
    phase = (frequency * time) % 1
    if phase < 0.5:
        return volume
    return -volume


def render_note(output, voice, note_name, start_time, duration, volume):
    frequency = note_frequency[note_name]
    start_index = int(start_time * sample_rate)
    num_samples = int(duration * sample_rate)

    for sample_index in range(num_samples):
        output_index = start_index + sample_index

        if output_index >= len(output):
            break

        output[output_index] += voice(sample_index, frequency, volume)

We can actually just generate the lookup table so it supports more notes. Then we dont have to write out the frequency for every note on the piano.

note_names = [
    "C", "C#", "D", "D#", "E", "F",
    "F#", "G", "G#", "A", "A#", "B",
]

# build the note freq lut
note_frequency = {}
for midi_number in range(21, 109):
    # MIDI note 69 is A4.
    # A4 is the usual tuning anchor, 440 cycles per second.
    semitones_from_a4 = midi_number - 69
    frequency = 440 * (2 ** (semitones_from_a4 / 12))

    note_name = note_names[midi_number % 12]
    octave = (midi_number // 12) - 1
    full_name = note_name + str(octave)

    note_frequency[full_name] = frequency


print(note_frequency["A4"])   # 440.0
print(note_frequency["C4"])   # 261.625565...
print(note_frequency["F#5"])  # 739.988845...
